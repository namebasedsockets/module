00036 /* Logically inserts a new node into the pending queue.  Must *not* be called
00037  * from an interrupt handler.
00038  * Literally it instead looks for a node with a reference count of 0 first, and
00039  * reclaims it if it finds one.  Otherwise it allocates a new node to insert
00040  * into the pending queue.  This mechanism allows pending_queue_find_and_remove
00041  * to avoid taking the queue lock, which is important since it's called from
00042  * an irq handler.
00043  */
00044 static int pending_queue_push(struct sk_buff *sk, __u32 seq,
00045                               void *cb, void *data)
00046 {
00047         struct pending_node *ptr, *node = NULL;
00048         int err = -ENOMEM, insert = 0;
00049 
00050         printk(KERN_INFO "pending queue is %p\n", pending_queue);
00051         lock_pending_queue();
00052         for (ptr = pending_queue, node = NULL; ptr && !node; ptr = ptr->next)
00053                 if (ptr->ref.counter == 0)
00054                 {
00055                         /* Found a node that needs to be freed, claim it */
00056                         ptr->ref.counter = 1;
00057                         node = ptr;
00058                 }
00059         unlock_pending_queue();
00060         if (!node)
00061         {
00062                 node = kmalloc(sizeof(struct pending_node), GFP_ATOMIC);
00063                 if (node)
00064                 {
00065                         node->ref.counter = 1;
00066                         insert = 1;
00067                 }
00068         }
00069         if (node)
00070         {
00071                 err = 0;
00072                 node->seq = seq;
00073                 node->cb = cb;
00074                 node->data = data;
00075                 if (insert)
00076                 {
00077                         lock_pending_queue();
00078                         node->next = pending_queue;
00079                         pending_queue = node;
00080                         unlock_pending_queue();
00081                 }
00082         }
00083         return err;
00084 }

00086 static struct pending_node *pending_queue_find_and_remove(__u32 seq)
00087 {
00088         struct pending_node *node = NULL;
00089 
00090         for (node = pending_queue; node; node = node->next)
00091                 if (atomic_read(&node->ref) && node->seq == seq)
00092                         break;
00093         if (node)
00094         {
00095                 /* "Free" the node by decrementing its reference counter.
00096                  * It'll actually get freed later, in pending_queue_free.
00097                  */
00098                 atomic_dec(&node->ref);
00099         }
00100         return node;
00101 }

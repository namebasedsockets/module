00037 /*%
00038  * Structure for query header.  The order of the fields is machine- and
00039  * compiler-dependent, depending on the byte/bit order and the layout
00040  * of bit fields.  We use bit fields only in int variables, as this
00041  * is all ANSI requires.  This requires a somewhat confusing rearrangement.
00042  */
00043 
00044 typedef struct {
00045         unsigned        id :16;         /*%< query identification number */
00046 #ifdef __BIG_ENDIAN
00047                         /* fields in third byte */
00048         unsigned        qr: 1;          /*%< response flag */
00049         unsigned        opcode: 4;      /*%< purpose of message */
00050         unsigned        aa: 1;          /*%< authoritive answer */
00051         unsigned        tc: 1;          /*%< truncated message */
00052         unsigned        rd: 1;          /*%< recursion desired */
00053                         /* fields in fourth byte */
00054         unsigned        ra: 1;          /*%< recursion available */
00055         unsigned        unused :1;      /*%< unused bits (MBZ as of 4.9.3a3) */
00056         unsigned        ad: 1;          /*%< authentic data from named */
00057         unsigned        cd: 1;          /*%< checking disabled by resolver */
00058         unsigned        rcode :4;       /*%< response code */
00059 #endif
00060 #ifdef __LITTLE_ENDIAN
00061                         /* fields in third byte */
00062         unsigned        rd :1;          /*%< recursion desired */
00063         unsigned        tc :1;          /*%< truncated message */
00064         unsigned        aa :1;          /*%< authoritive answer */
00065         unsigned        opcode :4;      /*%< purpose of message */
00066         unsigned        qr :1;          /*%< response flag */
00067                         /* fields in fourth byte */
00068         unsigned        rcode :4;       /*%< response code */
00069         unsigned        cd: 1;          /*%< checking disabled by resolver */
00070         unsigned        ad: 1;          /*%< authentic data from named */
00071         unsigned        unused :1;      /*%< unused bits (MBZ as of 4.9.3a3) */
00072         unsigned        ra :1;          /*%< recursion available */
00073 #endif
00074                         /* remaining bytes */
00075         unsigned        qdcount :16;    /*%< number of question entries */
00076         unsigned        ancount :16;    /*%< number of answer entries */
00077         unsigned        nscount :16;    /*%< number of authority entries */
00078         unsigned        arcount :16;    /*%< number of resource entries */
00079 } HEADER;
